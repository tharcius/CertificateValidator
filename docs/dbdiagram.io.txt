Table students {
  id int [pk, increment]
  name varchar (60)
  email varchar (60)
  created_at datetime
  updated_at datetime
}

Table courses {
  id int [pk, increment]
  name varchar (60)
  duration int
  description text
  created_at datetime
  updated_at datetime
}

Table course_student {
  id int [pk, increment]
  course_id int [ref: - courses.id]
  student_id int [ref: - students.id]

}

Table schools{
  id int [pk, increment]
  name varchar(60)
  logo varchar(60)
  certificate varchar(60)
}

Table certificates {
  id int [pk, increment]
  student_id int [ref: > students.id]
  course_id int [ref: > courses.id]
  school_id int [ref: > schools.id]
  viewed int [default: 0]
  conclusion_date datetime
  certificate_code varchar(10)
  created_at datetime
  updated_at datetime

  indexes{
    certificate_code
  }
}
