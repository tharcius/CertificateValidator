Table students {
  id int [pk, increment]
  name varchar (60)
  email varchar (60)
  created_at datetime
  updated_at datetime
}

Table course_syllabuses {
  id int [pk, increment]
  name varchar (60)
  program_details text
  created_at datetime
  updated_at datetime
}

Table courses {
  id int [pk, increment]
  course_program_id int [ref: - course_syllabuses.id]
  name varchar (60)
  hourly_load int
  created_at datetime
  updated_at datetime
}

Table schools{
  id int [pk, increment]
  name varchar(60)
  logo varchar(60)
  certificate varchar(60)
}

Table certificates {
  id int [pk, increment]
  student_id int [ref: > students.id]
  course_id int [ref: > courses.id]
  school_id int [ref: > schools.id]
  conclusion_date datetime
  certificate_code varchar(10)
  created_at datetime
  updated_at datetime
}
